```sql
CREATE DATABASE IF NOT EXISTS interview_lab
  DEFAULT CHARACTER SET utf8mb4
  DEFAULT COLLATE utf8mb4_0900_ai_ci;

USE interview_lab;

DROP TABLE IF EXISTS order_logs;
DROP TABLE IF EXISTS order_items;
DROP TABLE IF EXISTS orders;
DROP TABLE IF EXISTS products;
DROP TABLE IF EXISTS users;

CREATE TABLE users (
    id BIGINT PRIMARY KEY AUTO_INCREMENT COMMENT '用户ID，用户表主键',
    tenant_id BIGINT NOT NULL DEFAULT 1 COMMENT '租户ID，用于模拟多租户或大客户数据隔离',
    username VARCHAR(64) NOT NULL COMMENT '用户名，用户展示名称',
    phone VARCHAR(20) NOT NULL COMMENT '手机号，用于登录、查重和隐式类型转换实验',
    email VARCHAR(128) NOT NULL COMMENT '邮箱，用于唯一键冲突和软删除实验',
    status TINYINT NOT NULL DEFAULT 1 COMMENT '用户状态，1表示正常，0表示禁用',
    deleted TINYINT NOT NULL DEFAULT 0 COMMENT '软删除标记，0表示未删除，1表示已删除',
    city VARCHAR(64) NOT NULL COMMENT '城市，用于城市和年龄组合查询实验',
    age INT NOT NULL COMMENT '年龄，用于范围查询和组合索引实验',
    created_at DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',
    updated_at DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT '更新时间',

    UNIQUE KEY uk_phone (phone),
    UNIQUE KEY uk_email (email),
    KEY idx_tenant_status_created_at (tenant_id, status, created_at),
    KEY idx_email_deleted (email, deleted),
    KEY idx_city_age (city, age),
    KEY idx_status_created_at (status, created_at)
) ENGINE=InnoDB;

CREATE TABLE products (
    id BIGINT PRIMARY KEY AUTO_INCREMENT COMMENT '商品ID，商品表主键',
    tenant_id BIGINT NOT NULL DEFAULT 1 COMMENT '租户ID，用于模拟多租户商品数据',
    name VARCHAR(128) NOT NULL COMMENT '商品名称',
    category VARCHAR(64) NOT NULL COMMENT '商品分类，用于分类筛选和组合索引实验',
    price DECIMAL(10,2) NOT NULL COMMENT '商品价格，用于范围查询实验',
    stock INT NOT NULL DEFAULT 0 COMMENT '库存数量，用于库存更新和锁冲突实验',
    status TINYINT NOT NULL DEFAULT 1 COMMENT '商品状态，1表示上架，0表示下架',
    created_at DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',

    KEY idx_category_price (category, price),
    KEY idx_status_stock (status, stock)
) ENGINE=InnoDB;

CREATE TABLE orders (
    id BIGINT PRIMARY KEY AUTO_INCREMENT COMMENT '订单ID，订单表主键',
    tenant_id BIGINT NOT NULL DEFAULT 1 COMMENT '租户ID，用于模拟多租户和大客户订单数据',
    order_no VARCHAR(64) NOT NULL COMMENT '订单号，业务唯一标识',
    user_id BIGINT NOT NULL COMMENT '用户ID，关联 users.id',
    total_amount DECIMAL(10,2) NOT NULL COMMENT '订单总金额，用于金额范围查询实验',
    status TINYINT NOT NULL DEFAULT 0 COMMENT '订单状态，0表示待支付，1表示已支付，2表示已发货，3表示已完成，4表示已取消',
    pay_type TINYINT NOT NULL DEFAULT 0 COMMENT '支付方式，0表示未支付，1表示微信，2表示支付宝，3表示银行卡',
    source TINYINT NOT NULL DEFAULT 0 COMMENT '订单来源，0表示未知，1表示App，2表示Web，3表示小程序',
    deleted TINYINT NOT NULL DEFAULT 0 COMMENT '软删除标记，0表示未删除，1表示已删除',
    pay_time DATETIME NULL COMMENT '支付时间，未支付时为空',
    created_at DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间，用于订单列表排序和时间范围查询',
    updated_at DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT '更新时间',

    UNIQUE KEY uk_order_no (order_no),
    KEY idx_tenant_created_at (tenant_id, created_at),
    KEY idx_user_id_created_at (user_id, created_at),
    KEY idx_status_created_at (status, created_at),
    KEY idx_created_at (created_at)
) ENGINE=InnoDB;

CREATE TABLE order_items (
    id BIGINT PRIMARY KEY AUTO_INCREMENT COMMENT '订单明细ID，订单明细表主键',
    tenant_id BIGINT NOT NULL DEFAULT 1 COMMENT '租户ID，用于模拟多租户订单明细数据',
    order_id BIGINT NOT NULL COMMENT '订单ID，关联 orders.id',
    product_id BIGINT NOT NULL COMMENT '商品ID，关联 products.id',
    quantity INT NOT NULL COMMENT '购买数量',
    price DECIMAL(10,2) NOT NULL COMMENT '下单时商品单价',
    created_at DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',

    KEY idx_order_id (order_id),
    KEY idx_product_id (product_id),
    KEY idx_product_order (product_id, order_id)
) ENGINE=InnoDB;

CREATE TABLE order_logs (
    id BIGINT PRIMARY KEY AUTO_INCREMENT COMMENT '订单日志ID，订单日志表主键',
    tenant_id BIGINT NOT NULL DEFAULT 1 COMMENT '租户ID，用于模拟多租户订单日志数据',
    order_id BIGINT NOT NULL COMMENT '订单ID，关联 orders.id',
    old_status TINYINT NULL COMMENT '变更前订单状态，首次创建时可为空',
    new_status TINYINT NOT NULL COMMENT '变更后订单状态',
    remark VARCHAR(255) COMMENT '日志备注，记录状态变更原因',
    created_at DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',

    KEY idx_order_id_created_at (order_id, created_at),
    KEY idx_created_at (created_at)
) ENGINE=InnoDB;
```

## 设计说明

这份建表脚本是整套 MySQL 面试实验室的基础。所有后续实验都基于 `interview_lab` 数据库中的五张表展开，不在每个实验里重新设计表结构。

### users：用户表

`users` 用于用户维度查询实验。重点字段包括：

- `id`：用户主键 ID，用于精确定位单行数据。
- `tenant_id`：租户 ID，用于模拟多租户或大客户隔离。
- `phone`：手机号，建立唯一索引 `uk_phone`，用于点查和隐式类型转换实验。
- `email`：邮箱，建立唯一索引 `uk_email`，用于唯一键冲突和软删除组合讨论。
- `city`、`age`：建立联合索引 `idx_city_age`，用于等值条件加范围条件实验。
- `status`、`deleted`：用于状态筛选和软删除查询条件设计。

### products：商品表

`products` 用于商品查询、库存扣减和锁冲突实验。重点字段包括：

- `id`：商品主键 ID，固定商品 `products.id = 2001` 和 `products.id = 2002` 会在造数脚本中保留。
- `category`、`price`：建立联合索引 `idx_category_price`，用于分类加价格范围查询。
- `stock`：库存数量，用于并发扣减、行锁、死锁和锁等待实验。
- `status`：商品状态，用于低选择性字段和库存组合查询。

### orders：订单表

`orders` 是索引、事务、深分页和慢查询排查的核心表。重点字段包括：

- `id`：订单主键 ID，固定订单 `orders.id = 3001` 和 `orders.id = 3002` 会在造数脚本中保留。
- `tenant_id`：租户 ID，热点租户 `tenant_id = 1` 用于数据倾斜和慢查询实验。
- `order_no`：订单号，建立唯一索引 `uk_order_no`，用于业务唯一点查。
- `user_id`：用户 ID，关联 `users.id`，用于用户订单列表和深分页实验。
- `status`：订单状态，用于订单状态流转、低选择性字段索引和事务可见性实验。
- `created_at`：创建时间，用于范围查询、排序、分页和深分页实验。
- `pay_time`：支付时间，用于支付订单筛选和报表类查询。

脚本保留了 `idx_tenant_created_at`、`idx_user_id_created_at`、`idx_status_created_at`、`idx_created_at` 等基础索引，但故意没有直接创建 `orders(user_id, status, created_at, id)` 这类更贴近订单列表的覆盖索引，目的是给后续索引实验留下优化空间。

### order_items：订单明细表

`order_items` 用于订单与商品关联、JOIN 查询和销量聚合实验。重点字段包括：

- `order_id`：订单 ID，关联 `orders.id`，建立索引 `idx_order_id`，用于订单查明细。
- `product_id`：商品 ID，关联 `products.id`，建立索引 `idx_product_id`，用于商品销量聚合。
- `quantity`、`price`：用于销量、销售额聚合统计。
- `created_at`：创建时间，用于报表时间范围过滤。

当前脚本没有直接创建 `order_items(product_id, created_at)`，用于制造商品销量报表慢查询和索引优化空间。

### order_logs：订单日志表

`order_logs` 用于订单状态变更日志、日志深分页和时间范围扫描实验。重点字段包括：

- `order_id`：订单 ID，关联 `orders.id`，建立联合索引 `idx_order_id_created_at`，用于查询单个订单日志。
- `old_status`、`new_status`：状态变更前后值，用于观察订单状态流转。
- `created_at`：创建时间，建立索引 `idx_created_at`，用于日志时间范围查询和归档类慢查询实验。

### 设计原则

这套表结构不是为了“从一开始就最优”，而是为了面试训练：

1. 保留基础可用索引，让常见点查、列表查询可以跑起来。
2. 故意缺少部分最优联合索引，用来观察回表、filesort、深分页、扫描行数过大等问题。
3. 固定字段命名和固定数据 ID，方便后续实验题直接引用。
4. 不创建外键约束，避免造数和并发实验受外键检查影响；但字段注释中说明逻辑关联关系。
5. 所有表使用 InnoDB，用于支持事务、MVCC、行锁、间隙锁、临键锁和死锁实验。
